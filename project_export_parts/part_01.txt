import 'package:go_router/go_router.dart';
import 'package:syria_store/features/splash/presentation/splash_page.dart';
import 'package:syria_store/features/home/presentation/pages/home_page.dart';
import 'package:syria_store/features/products/presentation/pages/category_products_page.dart';
import 'package:syria_store/features/products/presentation/pages/product_details_page.dart';
import 'package:syria_store/features/cart/presentation/pages/cart_screen.dart';
import 'package:syria_store/features/admin/presentation/pages/admin_panel_page.dart';

final goRouter = GoRouter(
  initialLocation: '/splash',
  routes: [
    // المسارات التي لا تحتاج للقائمة الجانبية
    GoRoute(
      path: '/splash',
      builder: (context, state) => const SplashPage(),
    ),

    // المسارات العادية للتطبيق
    GoRoute(
      path: '/home',
      builder: (context, state) => const HomePage(),
    ),
    GoRoute(
      path: '/cart',
      builder: (context, state) => const CartScreen(),
    ),
    GoRoute(
      path: '/admin',
      builder: (context, state) => const AdminPanelPage(),
    ),
    GoRoute(
      path: '/category/:categoryId',
      builder: (context, state) {
        final categoryId = state.pathParameters['categoryId']!;
        final categoryName = state.extra as String? ?? 'Products';
        return CategoryProductsPage(categoryId: categoryId, categoryName: categoryName);
      },
    ),
    GoRoute(
      path: '/product/:productId',
      builder: (context, state) {
        final productId = state.pathParameters['productId']!;
        return ProductDetailsPage(productId: productId);
      },
    ),
  ],
);

import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:go_router/go_router.dart';
import 'package:syria_store/features/auth/presentation/providers/auth_providers.dart';
import 'package:syria_store/core/models/models.dart';

class SideBar extends ConsumerWidget {
  const SideBar({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final userAsyncValue = ref.watch(currentUserProvider);

    void navigate(String path, {bool useGo = false}) {
      Navigator.of(context).pop(); // إغلاق الـ Drawer أولاً
      if (useGo) {
        context.go(path);
      } else {
        context.push(path);
      }
    }

    return Drawer(
      // سنحتفظ بالخلفية الداكنة الجميلة
      backgroundColor: const Color(0xFF17203A),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          // استخدام UserAccountsDrawerHeader لسهولة الاستخدام
          userAsyncValue.when(
            data: (user) => UserAccountsDrawerHeader(
              accountName: Text(user?.name ?? "زائر", style: const TextStyle(fontWeight: FontWeight.bold)),
              accountEmail: Text(user?.email ?? "يرجى تسجيل الدخول"),
              currentAccountPicture: const CircleAvatar(
                backgroundColor: Colors.white70,
                child: Icon(Icons.person, color: Color(0xFF17203A)),
              ),
              decoration: BoxDecoration(
                color: Theme.of(context).primaryColor.withOpacity(0.8),
              ),
            ),
            loading: () => const DrawerHeader(child: Center(child: CircularProgressIndicator(color: Colors.white))),
            error: (err, stack) => const DrawerHeader(child: Text("خطأ في تحميل المستخدم")),
          ),
          
          // استخدام ListTile مع أيقونات قياسية
          ListTile(
            leading: const Icon(Icons.home_outlined, color: Colors.white),
            title: const Text('الرئيسية', style: TextStyle(color: Colors.white)),
            onTap: () => navigate('/home', useGo: true),
          ),
          ListTile(
            leading: const Icon(Icons.shopping_cart_outlined, color: Colors.white),
            title: const Text('سلتي', style: TextStyle(color: Colors.white)),
            onTap: () => navigate('/cart'),
          ),
          ListTile(
            leading: const Icon(Icons.settings_outlined, color: Colors.white),
            title: const Text('الإعدادات', style: TextStyle(color: Colors.white)),
            onTap: () {
              // TODO: Create and navigate to settings page
              Navigator.of(context).pop();
            },
          ),
          
          const Divider(color: Colors.white24, indent: 16, endIndent: 16),

          // لوحة تحكم المسؤول
          userAsyncValue.when(
            data: (user) {
              if (user?.isAdmin ?? false) {
                return ListTile(
                  leading: const Icon(Icons.admin_panel_settings_outlined, color: Colors.white),
                  title: const Text('لوحة التحكم', style: TextStyle(color: Colors.white)),
                  onTap: () => navigate('/admin', useGo: true),
                );
              }
              return const SizedBox.shrink();
            },
            loading: () => const SizedBox.shrink(),
            error: (e, s) => const SizedBox.shrink(),
          ),
        ],
      ),
    );
  }
}

import 'package:syria_store/features/suppliers/presentation/pages/supplier_agreements_page.dart';

final goRouter = GoRouter(
  initialLocation: '/splash',
  routes: [
    // المسارات التي لا تحتاج للقائمة الجانبية
    GoRoute(
      path: '/splash',
      builder: (context, state) => const SplashPage(),
    ),

    // المسارات العادية للتطبيق
    GoRoute(
      path: '/home',
      builder: (context, state) => const HomePage(),
    ),
    GoRoute(
      path: '/cart',
      builder: (context, state) => const CartScreen(),
    ),
    GoRoute(
      path: '/admin',
      builder: (context, state) => const AdminPanelPage(),
    ),
    GoRoute(
      path: '/category/:categoryId',
      builder: (context, state) {
        final categoryId = state.pathParameters['categoryId']!;
        final categoryName = state.extra as String? ?? 'Products';
        return CategoryProductsPage(categoryId: categoryId, categoryName: categoryName);
      },
    ),
    GoRoute(
      path: '/product/:productId',
      builder: (context, state) {
        final productId = state.pathParameters['productId']!;
        return ProductDetailsPage(productId: productId);
      },
    ),
    GoRoute(
      path: '/supplier-agreements',
      builder: (context, state) => const SupplierAgreementsPage(),

    ),
  ],
);

class SideBar extends ConsumerWidget {
  const SideBar({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final userAsyncValue = ref.watch(currentUserProvider);

    void navigate(String path, {bool useGo = false}) {
      Navigator.of(context).pop(); // إغلاق الـ Drawer أولاً
      if (useGo) {
        context.go(path);
      } else {
        context.push(path);
      }
    }

    return Drawer(
      // سنحتفظ بالخلفية الداكنة الجميلة
      backgroundColor: const Color(0xFF17203A),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          // استخدام UserAccountsDrawerHeader لسهولة الاستخدام
          userAsyncValue.when(
            data: (user) => UserAccountsDrawerHeader(
              accountName: Text(user?.name ?? "زائر", style: const TextStyle(fontWeight: FontWeight.bold)),
              accountEmail: Text(user?.email ?? "يرجى تسجيل الدخول"),
              currentAccountPicture: const CircleAvatar(
                backgroundColor: Colors.white70,
                child: Icon(Icons.person, color: Color(0xFF17203A)),
              ),
              decoration: BoxDecoration(
                color: Theme.of(context).primaryColor.withOpacity(0.8),
              ),
            ),
            loading: () => const DrawerHeader(child: Center(child: CircularProgressIndicator(color: Colors.white))),
            error: (err, stack) => const DrawerHeader(child: Text("خطأ في تحميل المستخدم")),
          ),
          
          // استخدام ListTile مع أيقونات قياسية
          ListTile(
            leading: const Icon(Icons.home_outlined, color: Colors.white),
            title: const Text('الرئيسية', style: TextStyle(color: Colors.white)),
            onTap: () => navigate('/home', useGo: true),
          ),
          ListTile(
            leading: const Icon(Icons.shopping_cart_outlined, color: Colors.white),
            title: const Text('سلتي', style: TextStyle(color: Colors.white)),
            onTap: () => navigate('/cart'),
          ),
          ListTile(
            leading: const Icon(Icons.settings_outlined, color: Colors.white),
            title: const Text('الإعدادات', style: TextStyle(color: Colors.white)),
            onTap: () {
              // TODO: Create and navigate to settings page
              Navigator.of(context).pop();
            },
          ),
          
          const Divider(color: Colors.white24, indent: 16, endIndent: 16),

          // لوحة تحكم المسؤول
          userAsyncValue.when(
            data: (user) {
              if (user?.isAdmin ?? false) {

import 'package:syria_store/features/suppliers/presentation/pages/supplier_agreements_page.dart';

final goRouter = GoRouter(
  initialLocation: '/supplier-agreements',
  routes: [
    // --- المسار الرئيسي لقائمة الاتفاقيات مع مسار فرعي للتفاصيل ---
    GoRoute(
      path: '/supplier-agreements',
      builder: (context, state) => const SupplierAgreementsPage(),
      routes: [
        GoRoute(
          // لاحظ أن المسار نسبي الآن (بدون / في البداية)
          path: 'details/:agreementId',
          builder: (context, state) {
            final agreementId = state.pathParameters['agreementId']!;
            return AgreementDetailsPage(agreementId: agreementId);
          },
        ),
      ],
    ),
    // --- المسار الجديد لصفحة تفاصيل المورد ---
    GoRoute(
      path: '/suppliers/:supplierId',
      builder: (context, state) {
        // استلام اسم المورد كـ extra data لتحسين تجربة المستخدم
        final supplierName = state.extra as String? ?? 'تفاصيل المورد';
        final supplierId = state.pathParameters['supplierId']!;
        return SupplierDetailsPage(
          supplierId: supplierId,

          supplierName: supplierName,
        );
      },
    ),
    GoRoute(
      path: '/add-agreement',
      builder: (context, state) => const AddAgreementPage(),
    ),
    GoRoute(path: '/logs', builder: (context, state) => const LogsPage()),
  ],
);
